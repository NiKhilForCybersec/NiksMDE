KQL Queries for Metrics

KQL - SOC Metrics 
// Alert volume by severity (last 30 days)
AlertInfo
| where Timestamp > ago(30d)
| summarize Count = count() by Severity
| order by Count desc

// Daily alert trend
AlertInfo
| where Timestamp > ago(30d)
| summarize DailyAlerts = count() by bin(Timestamp, 1d)
| render timechart

// Alerts by detection source
AlertInfo
| where Timestamp > ago(30d)
| summarize Count = count() by DetectionSource
| order by Count desc

// Top alerting devices
AlertEvidence
| where Timestamp > ago(30d)
| where EntityType == "Machine"
| summarize AlertCount = dcount(AlertId) by DeviceId
| order by AlertCount desc
| take 10

// MITRE ATT&CK coverage
AlertInfo
| where Timestamp > ago(30d)
| mv-expand AttackTechniques
| summarize Count = count() by tostring(AttackTechniques)
| order by Count desc
