Detection Tuning

KQL - Detection Effectiveness 
// Alert volume by classification
AlertInfo
| where Timestamp > ago(30d)
| summarize 
    Total = count(),
    TP = countif(Classification == "TruePositive"),
    FP = countif(Classification == "FalsePositive"),
    Unclassified = countif(isempty(Classification))
| extend TPRate = round(100.0 * TP / Total, 1),
         FPRate = round(100.0 * FP / Total, 1)

// Noisiest detections (candidates for tuning)
AlertInfo
| where Timestamp > ago(30d)
| summarize 
    Alerts = count(),
    FP = countif(Classification == "FalsePositive") by Title
| extend FPRate = round(100.0 * FP / Alerts, 1)
| where Alerts > 10
| order by FPRate desc

// Detection coverage trend
AlertInfo
| where Timestamp > ago(90d)
| summarize Alerts = count() by bin(Timestamp, 7d)
| order by Timestamp
