KQL for Action Tracking

KQL - Action Queries 
// Recent remediation actions
DeviceEvents
| where Timestamp > ago(7d)
| where ActionType has_any ("Quarantine", "Remediation", "Blocked")
| summarize Count = count() by ActionType
| order by Count desc

// Files quarantined in last 24 hours
DeviceEvents
| where Timestamp > ago(24h)
| where ActionType == "AntivirusDetection"
| extend Fields = parse_json(AdditionalFields)
| where Fields.Action == "Quarantine"
| project Timestamp, DeviceName, FileName, Fields.ThreatName
