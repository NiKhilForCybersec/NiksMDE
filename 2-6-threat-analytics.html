Hunting Queries from Threat Analytics

KQL - Using Threat Report Queries 
// Threat reports include ready-to-use KQL queries
// Example: Hunt for ransomware indicators

// Look for file hashes from threat report
let malwareHashes = dynamic([
    "hash1...",
    "hash2...",
    "hash3..."
]);
DeviceFileEvents
| where Timestamp > ago(30d)
| where SHA256 in (malwareHashes)
| project Timestamp, DeviceName, FileName, FolderPath, SHA256

// Look for C2 communication
let c2Domains = dynamic([
    "bad-domain1.com",
    "bad-domain2.com"
]);
DeviceNetworkEvents
| where Timestamp > ago(30d)
| where RemoteUrl has_any (c2Domains)
| project Timestamp, DeviceName, RemoteUrl, InitiatingProcessFileName

// Look for specific attack technique
DeviceProcessEvents
| where Timestamp > ago(30d)
| where ProcessCommandLine has_any (
    "specific-tool",
    "attack-indicator"
)
| project Timestamp, DeviceName, FileName, ProcessCommandLine

// Check mitigation coverage
DeviceTvmSecureConfigurationAssessment
| where ConfigurationId == "specific-mitigation-id"
| summarize 
    Compliant = countif(IsCompliant == true),
    NonCompliant = countif(IsCompliant == false)
| extend ComplianceRate = Compliant * 100.0 / (Compliant + NonCompliant)
