API Method

PowerShell - API Collection 
# Initiate collection
$machineId = "device-guid"
$body = @{
    Comment = "Forensic collection for incident investigation"
} | ConvertTo-Json

$uri = "https://api.securitycenter.microsoft.com/api/machines/$machineId/collectInvestigationPackage"
$response = Invoke-RestMethod -Method Post -Uri $uri -Headers $headers -Body $body

# Check status
$actionId = $response.id
$statusUri = "https://api.securitycenter.microsoft.com/api/machineactions/$actionId"

do {
    $status = Invoke-RestMethod -Method Get -Uri $statusUri -Headers $headers
    Write-Host "Status: $($status.status)"
    Start-Sleep -Seconds 30
} while ($status.status -notin @("Succeeded", "Failed"))

# Get download URI
$packageUri = "https://api.securitycenter.microsoft.com/api/machines/$machineId/getPackageUri"
$downloadInfo = Invoke-RestMethod -Method Get -Uri $packageUri -Headers $headers

# Download
Invoke-WebRequest -Uri $downloadInfo.value -OutFile "investigation_package.zip"

Automated Analysis Script

PowerShell - Quick Analysis 
# Quick investigation package analysis
$packagePath = "C:\Forensics\InvestigationPackage"

# Check autoruns for unsigned entries
Import-Csv "$packagePath\Autoruns\Autoruns.csv" |
    Where-Object { $_.Publisher -eq "" -or $_.Publisher -eq "(Not verified)" } |
    Select-Object "Entry Location", Entry, "Image Path"

# Check processes from unusual locations
Import-Csv "$packagePath\Processes\Processes.csv" |
    Where-Object { 
        $_.Path -like "*\Temp\*" -or 
        $_.Path -like "*\Downloads\*" -or
        $_.Path -like "*\Public\*"
    } |
    Select-Object "Process Name", Path, "Command Line"

# Check external network connections
Import-Csv "$packagePath\NetworkConnections\NetworkConnections.csv" |
    Where-Object { 
        $_."Remote Address" -notlike "10.*" -and
        $_."Remote Address" -notlike "192.168.*" -and
        $_."Remote Address" -notlike "172.16.*" -and
        $_.State -eq "ESTABLISHED"
    } |
    Select-Object "Process Name", "Remote Address", "Remote Port"
