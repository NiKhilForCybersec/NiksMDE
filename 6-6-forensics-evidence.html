LIVE RESPONSE EVIDENCE COLLECTION:

COLLECT SPECIFIC FILES:
# Download suspicious file
getfile "C:\Users\victim\AppData\Local\Temp\malware.exe"

# Collect browser history
getfile "C:\Users\victim\AppData\Local\Google\Chrome\User Data\Default\History"

# Collect registry hive
getfile "C:\Windows\System32\config\SAM"

COLLECT MEMORY:
# Full memory dump (requires library upload)
run memdump.exe -o memory.dmp

# Process-specific memory
run procdump.exe -ma [PID] process.dmp

COLLECT REGISTRY:
# Export specific keys
registry export HKLM\SOFTWARE\Microsoft\Windows\CurrentVersion\Run run_keys.reg

# Query specific value
registry query HKLM\SOFTWARE\Microsoft\Windows\CurrentVersion\Run

COLLECT EVENT LOGS:
# Export specific log
run wevtutil.exe epl Security C:\evidence\security.evtx

# Query recent events
run wevtutil.exe qe Security /c:100 /rd:true /f:text

COLLECT NETWORK STATE:
# Current connections
run netstat.exe -anob > connections.txt

# DNS cache
run ipconfig.exe /displaydns > dns_cache.txt

# ARP table
run arp.exe -a > arp_table.txt

COLLECT PROCESS INFO:
# Process list with details
run tasklist.exe /v > processes.txt

# Process with command lines
run wmic.exe process get name,processid,commandline > process_cmdlines.txt

Timeline Reconstruction

KQL - Forensic Timeline 
// Build comprehensive timeline for device
let TargetDevice = "WKSTN-HR-042";
let StartTime = datetime(2024-01-15 08:00:00);
let EndTime = datetime(2024-01-15 18:00:00);
// Process events
let ProcessEvents = 
    DeviceProcessEvents
    | where DeviceName == TargetDevice
    | where Timestamp between (StartTime .. EndTime)
    | extend EventType = "Process"
    | project Timestamp, EventType, 
             Details = strcat(ActionType, ": ", FileName);
// File events
let FileEvents = 
    DeviceFileEvents
    | where DeviceName == TargetDevice
    | where Timestamp between (StartTime .. EndTime)
    | extend EventType = "File"
    | project Timestamp, EventType,
             Details = strcat(ActionType, ": ", FileName);
// Network events
let NetworkEvents = 
    DeviceNetworkEvents
    | where DeviceName == TargetDevice
    | where Timestamp between (StartTime .. EndTime)
    | extend EventType = "Network"
    | project Timestamp, EventType,
             Details = strcat(ActionType, ": ", RemoteIP);
// Combine all events
union ProcessEvents, FileEvents, NetworkEvents
| order by Timestamp asc
