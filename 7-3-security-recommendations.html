KQL Queries

KQL - Security Recommendations 
// Devices missing critical configurations
DeviceTvmSecureConfigurationAssessment
| where IsCompliant == 0
| where IsApplicable == 1
| summarize NonCompliantDevices = dcount(DeviceId) by ConfigurationId
| order by NonCompliantDevices desc

// ASR rule compliance
DeviceTvmSecureConfigurationAssessment
| where ConfigurationCategory == "AttackSurfaceReduction"
| summarize 
    Compliant = countif(IsCompliant == 1),
    NonCompliant = countif(IsCompliant == 0)
    by ConfigurationId

// Software needing updates
DeviceTvmSoftwareInventory
| where NumberOfWeaknesses > 0
| summarize 
    Vulnerabilities = sum(NumberOfWeaknesses),
    Devices = dcount(DeviceId)
    by SoftwareName, SoftwareVersion
| order by Vulnerabilities desc
